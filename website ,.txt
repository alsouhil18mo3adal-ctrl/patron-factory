  <script>
    (function () {
      'use strict';

      const noMotion = window.matchMedia('(prefers-reduced-motion: reduce)').matches;

      /* ── Mobile Nav ── */
      const toggle = document.getElementById('menu-toggle');
      const menu   = document.getElementById('menu-links');
      if (toggle && menu) {
        toggle.addEventListener('click', () => {
          const open = menu.classList.toggle('open');
          toggle.classList.toggle('open');
          toggle.setAttribute('aria-expanded', open);
        });
        menu.querySelectorAll('a').forEach(a => a.addEventListener('click', () => {
          menu.classList.remove('open');
          toggle.classList.remove('open');
          toggle.setAttribute('aria-expanded', 'false');
        }));
      }

      /* ── Nav Scroll ── */
      const nav = document.querySelector('nav');
      window.addEventListener('scroll', () => nav.classList.toggle('scrolled', scrollY > 50), { passive: true });

      /* ── Canvas Background ── */
      if (!noMotion) {
        const cv = document.getElementById('bg-canvas');
        if (cv) {
          const cx = cv.getContext('2d');
          let W = cv.width  = innerWidth;
          let H = cv.height = innerHeight;
          const TC = 42, MP = 72;
          const threads = Array.from({ length: TC }, (_, i) => ({
            x: (W / TC) * i + Math.random() * 15,
            op: 0.03 + Math.random() * 0.1,
            sp: 0.003 + Math.random() * 0.006,
            amp: 14 + Math.random() * 22,
            off: Math.random() * 100,
            col: Math.random() > 0.65 ? '#C49A3C' : '#F5F0E8'
          }));
          class P {
            constructor() { this.reset(); this.y = Math.random() * H; }
            reset() { this.x = Math.random()*W; this.y=H+10; this.r=1+Math.random()*1.5; this.sp=0.15+Math.random()*0.45; this.op=0.15+Math.random()*0.35; this.td=1; this.tf=0.01+Math.random()*0.02; }
            upd() { this.y-=this.sp; this.op+=this.tf*this.td; if(this.op>0.6)this.td=-1; else if(this.op<0.15)this.td=1; if(this.y<-10)this.reset(); }
            drw() { cx.beginPath(); cx.arc(this.x,this.y,this.r,0,Math.PI*2); cx.fillStyle='#C49A3C'; cx.globalAlpha=this.op; cx.fill(); }
          }
          const pts = Array.from({ length: MP }, () => new P());
          window.addEventListener('resize', () => {
            W=cv.width=innerWidth; H=cv.height=innerHeight;
            threads.forEach((t,i)=>t.x=(W/TC)*i+Math.random()*15);
          }, { passive: true });
          (function anim() {
            cx.clearRect(0,0,W,H);
            threads.forEach(t => {
              t.off+=t.sp; cx.beginPath(); cx.lineWidth=0.8; cx.strokeStyle=t.col; cx.globalAlpha=t.op;
              cx.moveTo(t.x,0);
              for(let y=0;y<=H;y+=40) cx.lineTo(t.x+Math.sin((y*0.0015)+t.off)*t.amp,y);
              cx.stroke();
            });
            pts.forEach(p=>{p.upd();p.drw();}); cx.globalAlpha=1; requestAnimationFrame(anim);
          })();
        }
      }

      /* ── Hero Tilt ── */
      if (!noMotion) {
        const ht = document.getElementById('hero-tilt');
        const hi = document.getElementById('hero-tilt-inner');
        if (ht && hi) {
          ht.addEventListener('mousemove', e => {
            const r = ht.getBoundingClientRect();
            hi.style.transform = `rotateX(${((e.clientY-r.top-r.height/2)/(r.height/2))*-5}deg) rotateY(${((e.clientX-r.left-r.width/2)/(r.width/2))*5}deg)`;
          });
          ht.addEventListener('mouseleave', () => hi.style.transform = 'rotateX(0deg) rotateY(0deg)');
        }
      }

      /* ── Scroll Reveal + Stat Counters ── */
      let counted = false;
      const reveals = document.querySelectorAll('.reveal');
      if (reveals.length) {
        const obs = new IntersectionObserver(entries => {
          entries.forEach(e => {
            if (!e.isIntersecting) return;
            e.target.classList.add('visible');
            if (e.target.classList.contains('about-left') && !counted) {
              counted = true;
              count(document.getElementById('stat-wilayas'), 0, 69, 1800, '+');
              count(document.getElementById('stat-years'),   0, 10, 1600, '+');
            }
            obs.unobserve(e.target);
          });
        }, { threshold: 0.14, rootMargin: '0px 0px -50px 0px' });
        reveals.forEach(el => obs.observe(el));
      }

      function count(el, s, e, dur, sfx) {
        if (!el) return;
        let t0;
        (function step(ts) {
          if (!t0) t0 = ts;
          const p = Math.min((ts - t0) / dur, 1);
          el.textContent = Math.floor(p * (e - s) + s) + (sfx || '');
          if (p < 1) requestAnimationFrame(step);
        })(performance.now());
      }

      /* ── 4-Step Timeline Stitch ── */
      const proc    = document.getElementById('process');
      const bgP     = document.querySelector('.tl-bg');
      const animP   = document.getElementById('timeline-stitch');
      const badges  = document.querySelectorAll('.step-badge');

      function buildPath() {
        if (!proc || !bgP || !animP || badges.length < 4) return;
        const cr = document.querySelector('.timeline-container').getBoundingClientRect();
        const pts = Array.from(badges).map(b => {
          const r = b.getBoundingClientRect();
          return { x: r.left + r.width/2 - cr.left, y: r.top + r.height/2 - cr.top };
        });
        let d;
        if (innerWidth <= 768) {
          d = `M ${pts[0].x} ${pts[0].y} Q ${pts[0].x+30} ${(pts[0].y+pts[1].y)/2}, ${pts[1].x} ${pts[1].y}
               Q ${pts[1].x-30} ${(pts[1].y+pts[2].y)/2}, ${pts[2].x} ${pts[2].y}
               Q ${pts[2].x+30} ${(pts[2].y+pts[3].y)/2}, ${pts[3].x} ${pts[3].y}`;
        } else {
          d = `M ${pts[0].x} ${pts[0].y}
               Q ${(pts[0].x+pts[1].x)/2} ${pts[0].y-22}, ${pts[1].x} ${pts[1].y}
               Q ${(pts[1].x+pts[2].x)/2} ${pts[1].y+22}, ${pts[2].x} ${pts[2].y}
               Q ${(pts[2].x+pts[3].x)/2} ${pts[2].y-22}, ${pts[3].x} ${pts[3].y}`;
        }
        bgP.setAttribute('d', d); animP.setAttribute('d', d);
        const len = animP.getTotalLength();
        animP.style.strokeDasharray  = len;
        animP.style.strokeDashoffset = len;
        window._tlLen = len;
        drawPath();
      }
      function drawPath() {
        if (!proc || !animP || !window._tlLen) return;
        const sr = proc.getBoundingClientRect();
        const p  = Math.max(0, Math.min(1, -(sr.top - innerHeight + 150) / (sr.height - 150)));
        animP.style.strokeDashoffset = window._tlLen * (1 - p);
      }
      if (proc && animP) {
        window.addEventListener('load', buildPath);
        window.addEventListener('resize', buildPath);
        window.addEventListener('scroll', drawPath, { passive: true });
        setTimeout(buildPath, 120);
      }

      /* ── Gallery Tilt ── */
      if (!noMotion) {
        document.querySelectorAll('[data-tilt-gallery]').forEach(c => {
          c.style.transformOrigin = 'center center';
          c.addEventListener('mousemove', e => {
            const r = c.getBoundingClientRect();
            c.style.transform = `perspective(1200px) rotateX(${((e.clientY-r.top-r.height/2)/(r.height/2))*-10}deg) rotateY(${((e.clientX-r.left-r.width/2)/(r.width/2))*10}deg)`;
          });
          c.addEventListener('mouseleave', () => c.style.transform = 'perspective(1200px) rotateX(0deg) rotateY(0deg)');
        });
      }

      /* ── Contact Form ── */
      const form = document.getElementById('artex-contact-form');
      const stat = document.getElementById('form-response-status');
      if (form && stat) {
        form.querySelectorAll('.fc').forEach(f => {
          ['input','change'].forEach(ev => f.addEventListener(ev, () => { if (f.checkValidity()) f.classList.remove('invalid'); }));
        });
        form.addEventListener('submit', e => {
          e.preventDefault();
          if (!form.checkValidity()) {
            form.querySelectorAll('.fc').forEach(f => f.classList.toggle('invalid', !f.checkValidity()));
            stat.className = 'form-status error';
            stat.textContent = 'Veuillez remplir tous les champs obligatoires.';
            stat.style.display = 'block';
            return;
          }
          form.querySelectorAll('.fc').forEach(f => f.classList.remove('invalid'));
          stat.className = 'form-status';
          stat.textContent = 'Envoi en cours...';
          stat.style.display = 'block';
          setTimeout(() => {
            stat.className = 'form-status success';
            stat.textContent = 'Merci ! Votre demande a été reçue par notre atelier à Annaba. Nous vous contacterons sous 24h.';
            form.reset();
          }, 1500);
        });
      }

      /* ── Language Toggle ── */
      const langBtns = document.querySelectorAll('.lang-btn');
      let currentLang = 'fr';
      
      langBtns.forEach(btn => {
        btn.addEventListener('click', () => {
          const lang = btn.getAttribute('data-lang');
          currentLang = lang;
          langBtns.forEach(b => b.classList.remove('active'));
          btn.classList.add('active');
          
          if (lang === 'ar') {
            document.documentElement.lang = 'ar';
            document.documentElement.dir = 'rtl';
          } else {
            document.documentElement.lang = 'fr';
            document.documentElement.dir = 'ltr';
          }
          
          console.log('Language switched to:', lang);
        });
      });

    })();
